\section{工具与环境介绍}

\begin{frame}{演讲与展示工具概览}
    \textbf{选择合适的工具提升展示效果}

    \vspace{0.3cm}

    \begin{table}
        \centering
        \small
        \begin{tabular}{llp{5cm}}
            \toprule
            \textbf{工具类型} & \textbf{推荐工具} & \textbf{适用场景} \\
            \midrule
            幻灯片制作 & PowerPoint/Keynote & 通用场景，易上手 \\
            在线协作 & Google Slides & 团队协作，云端访问 \\
            创意展示 & Prezi & 非线性叙事，创意展示 \\
            设计美化 & Canva & 快速美化，模板丰富 \\
            学术演示 & LaTeX Beamer & 学术场合，公式多 \\
            代码演示 & Reveal.js & 技术分享，代码多 \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3cm}

    \textbf{工具选择建议：}
    \begin{itemize}
        \item 熟悉的工具 > 新潮的工具
        \item 简单的工具 > 复杂的工具
        \item 内容 > 形式：工具是手段，内容是核心
    \end{itemize}
\end{frame}

\begin{frame}{PowerPoint/Keynote使用技巧}
    \textbf{最常用工具的高效用法}

    \vspace{0.3cm}

    \textbf{效率提升技巧：}
    \begin{itemize}
        \item \textbf{母版设计}：统一风格，一次设置
        \item \textbf{样式复用}：格式刷快速复制格式
        \item \textbf}快捷键}：
            \begin{itemize}
                \item Ctrl+D：快速复制
                \item F5：从头放映
                \item Shift+F5：从当前页放映
            \end{itemize}
        \item \textbf{模板库}：善用专业模板
    \end{itemize}

    \vspace{0.3cm}

    \textbf{设计原则：}
    \begin{itemize}
        \item 一页一个观点
        \item 大字体（24pt+）
        \item 高质量图片
        \item 留白很重要
    \end{itemize}
\end{frame}

\begin{frame}{LaTeX Beamer简介}
    \textbf{学术和技术演讲的专业选择}

    \vspace{0.3cm}

    \textbf{Beamer的优势：}
    \begin{itemize}
        \item \textbf{专业排版}：学术级别的排版质量
        \item \textbf{公式支持}：完美的数学公式
        \item \textbf{代码高亮}：专业的代码展示
        \item \textbf{版本控制}：纯文本，便于版本管理
        \item \textbf{结构化}：强制结构化思维
    \end{itemize}

    \vspace{0.3cm}

    \textbf{Beamer的使用场景：}
    \begin{itemize}
        \item 学术报告
        \item 技术演讲
        \item 论文答辩
        \item 包含大量公式和代码的演讲
    \end{itemize}

    \vspace{0.3cm}

    \begin{block}{本课程使用Beamer}
    本课程的所有幻灯片都是用LaTeX Beamer制作的
    \end{block}
\end{frame}
