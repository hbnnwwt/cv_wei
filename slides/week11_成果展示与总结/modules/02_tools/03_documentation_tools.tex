\begin{frame}{文档与协作工具}
    \textbf{高效的工具提升团队协作效率}

    \vspace{0.3cm}

    \textbf{知识管理工具：}
    \begin{itemize}
        \item \textbf{Notion}：一体化工作空间，支持文档、数据库、看板
        \item \textbf{Confluence}：企业级知识管理，与Jira集成
        \item \textbf{语雀/飞书文档}：国内协作工具，适合中文环境
    \end{itemize}

    \vspace{0.2cm}

    \textbf{文档编写工具：}
    \begin{itemize}
        \item \textbf{Markdown + Typora}：轻量级，专注写作
        \item \textbf{VS Code}：代码+文档一体化
        \item \textbf{Sphinx}：自动生成API文档
    \end{itemize}

    \vspace{0.2cm}

    \textbf{代码托管工具：}
    \begin{itemize}
        \item \textbf{GitHub/GitLab}：版本控制+协作
        \item \textbf{Gitee}：国内平台，速度快
    \end{itemize}

    \vspace{0.2cm}

    \textbf{设计协作工具：}
    \begin{itemize}
        \item \textbf{Figma}：在线设计，实时协作
        \item \textbf{Draw.io}：流程图、架构图
    \end{itemize}
\end{frame}

\begin{frame}{Markdown快速入门}
    \textbf{轻量级标记语言}

    \vspace{0.3cm}

    \begin{columns}
        \column{0.5\textwidth}
        \textbf{基本语法：}
        \begin{itemize}
            \item \# 标题
            \item **粗体** / *斜体*
            \item - 列表
            \item \`代码\`
            \item [链接](url)
            \item ![图片](url)
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{工具推荐：}
        \begin{itemize}
            \item Typora：所见即所得
            \item VS Code + 插件
            \item Obsidian：知识管理
            \item 在线编辑器
        \end{itemize}
    \end{columns}

    \vspace{0.3cm}

    \begin{exampleblock}{Markdown示例}
    \small
    \# 项目标题\\
    这是项目的简介，支持**粗体**和*斜体*。\\
    \`\`\`python\\
    print("Hello, World!")\\
    \`\`\`
    \end{exampleblock}
\end{frame}
