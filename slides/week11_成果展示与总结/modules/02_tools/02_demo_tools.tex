\begin{frame}{OBS Studio快速入门}
    \textbf{最流行的开源录屏工具}

    \vspace{0.3cm}

    \textbf{OBS的特点：}
    \begin{itemize}
        \item 免费开源
        \item 功能强大（录制+直播）
        \item 跨平台（Windows/Mac/Linux）
        \item 支持多源混流
    \end{itemize}

    \vspace{0.3cm}

    \textbf{基本设置：}
    \begin{enumerate}
        \item 添加源：屏幕捕获、窗口捕获
        \item 设置输出：视频格式、质量
        \item 开始录制：点击开始录制
        \item 结束录制：保存视频文件
    \end{enumerate}

    \vspace{0.3cm}

    \textbf{演示录制建议：}
    \begin{itemize}
        \item 分辨率：1920x1080或1280x720
        \item 帧率：30fps足够
        \item 格式：MP4便于播放
    \end{itemize}
\end{frame}
