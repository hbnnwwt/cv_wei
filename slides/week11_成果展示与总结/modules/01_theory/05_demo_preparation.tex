\section{Demo演示技巧}

\begin{frame}{Demo准备：演示流程设计}
    \textbf{精心设计的演示流程让Demo更流畅}

    \vspace{0.3cm}

    \textbf{Demo流程设计原则：}
    \begin{enumerate}
        \item \textbf{完整但不冗长}：
            \begin{itemize}
                \item 展示完整的用户流程
                \item 跳过重复性操作
                \item 聚焦核心功能
            \end{itemize}

        \item \textbf{有故事线}：
            \begin{itemize}
                \item 从问题出发
                \item 展示解决方案
                \item 证明有效可行
            \end{itemize}

        \item \textbf{突出亮点}：
            \begin{itemize}
                \item 提前规划展示哪些功能
                \item 准备对比场景
                \item 设计"wow时刻"
            \end{itemize}

        \item \textbf{有备选方案}：
            \begin{itemize}
                \item 准备数据集
                \item 录制备份视频
                \item 准备截图PPT
            \end{itemize}
    \end{enumerate}
\end{frame}

\begin{frame}{Demo准备：场景与数据准备}
    \textbf{合适的测试数据让演示更有说服力}

    \vspace{0.3cm}

    \textbf{测试数据准备：}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{正常场景}
            \begin{itemize}
                \item 标准答题卡
                \item 清晰填涂
                \item 典型使用场景
            \end{itemize}
        \end{block}

        \column{0.5\textwidth}
        \begin{block}{边缘场景}
            \begin{itemize}
                \item 轻微倾斜
                \item 光照不均
                \item 部分填涂
            \end{itemize}
        \end{block}
    \end{columns}

    \vspace{0.3cm}

    \textbf{数据准备检查清单：}
    \begin{itemize}
        \item[$\square$] 准备至少5-10组测试数据
        \item[$\square$] 覆盖不同场景
        \item[$\square$] 提前验证能正确处理
        \item[$\square$] 准备"对比"数据（前后对比）
        \item[$\square$] 准备"失败"案例展示处理能力
    \end{itemize}

    \vspace{0.3cm}

    \begin{alertblock}{注意}
    使用真实数据比模拟数据更有说服力
    \end{alertblock}
\end{frame}

\begin{frame}{Demo准备：备用方案（Plan B）}
    \textbf{做好最坏打算，确保演示顺利进行}

    \vspace{0.3cm}

    \textbf{常见故障与备用方案：}
    \begin{table}
        \centering
        \small
        \begin{tabular}{ll}
            \toprule
            \textbf{可能的故障} & \textbf{备用方案} \\
            \midrule
            代码运行报错 & 切换到备份代码分支 \\
            网络连接失败 & 使用本地数据/离线模式 \\
            处理速度太慢 & 使用预处理好的数据 \\
            识别效果不佳 & 选择最理想的测试数据 \\
            演示环境异常 & 切换到录制的演示视频 \\
            投影设备问题 & 使用截图PPT继续讲解 \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3cm}

    \textbf{备用方案准备清单：}
    \begin{itemize}
        \item[$\square$] 录制完整的演示视频
        \item[$\square$] 准备截图PPT
        \item[$\square$] 准备多个测试数据集
        \item[$\square$] 准备备用代码版本
    \end{itemize}
\end{frame}
