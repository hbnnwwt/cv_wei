\begin{frame}{Demo执行：开场介绍}
    \textbf{好的开场让Demo更有意义}

    \vspace{0.3cm}

    \textbf{Demo开场三要素：}
    \begin{enumerate}
        \item \textbf{场景介绍}：
            \begin{itemize}
                \item "我们将演示如何处理这张答题卡..."
                \item "这是一个典型的XX场景..."
            \end{itemize}

        \item \textbf{预期结果}：
            \begin{itemize}
                \item "系统将自动识别出答案..."
                \item "这个过程大约需要X秒..."
            \end{itemize}

        \item \textbf{流程预告}：
            \begin{itemize}
                \item "演示将分为三个步骤..."
                \item "首先我们做X，然后Y，最后Z..."
            \end{itemize}
    \end{enumerate}

    \vspace{0.3cm}

    \begin{exampleblock}{Demo开场示例}
    \small
    "现在我来演示系统的核心功能。这是一张数学答题卡，包含20道选择题。我们的系统将自动定位答题卡位置，识别每个选项的填涂情况，最终输出答案和分数。整个过程大约需要3秒钟。"
    \end{exampleblock}
\end{frame}

\begin{frame}{Demo执行：功能展示顺序}
    \textbf{合理的展示顺序让演示更流畅}

    \vspace{0.3cm}

    \textbf{推荐的Demo展示顺序：}
    \begin{enumerate}
        \item \textbf{端到端演示}（30秒）：
            \begin{itemize}
                \item 先展示完整流程
                \item 让观众看到最终效果
                \item 建立整体印象
            \end{itemize}

        \item \textbf{分步骤演示}（2分钟）：
            \begin{itemize}
                \item 慢速展示关键步骤
                \item 解释每个环节的作用
                \item 展示中间结果
            \end{itemize}

        \item \textbf{亮点展示}（30秒）：
            \begin{itemize}
                \item 展示特殊功能
                \item 对比展示效果
                \item 强调创新点
            \end{itemize}

        \item \textbf{结果展示}（30秒）：
            \begin{itemize}
                \item 清晰展示最终结果
                \item 与预期对比
                \item 总结演示内容
            \end{itemize}
    \end{enumerate}
\end{frame}

\begin{frame}{Demo执行：实时问题处理}
    \textbf{当Demo出问题时保持专业}

    \vspace{0.3cm}

    \textbf{技术故障应对原则：}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{block}{保持冷静}
            \begin{itemize}
                \item 不要慌张
                \item 不要反复尝试
                \item 保持微笑
            \end{itemize}
        \end{block}

        \column{0.5\textwidth}
        \begin{block}{快速切换}
            \begin{itemize}
                \item 立即启动Plan B
                \item 平滑过渡
                \item 不要浪费时间
            \end{itemize}
        \end{block}
    \end{columns}

    \vspace{0.3cm}

    \textbf{应对话术示例：}
    \begin{itemize}
        \item "看来我们的AI今天有点害羞，让我用录制好的视频来展示..."
        \item "由于现场网络限制，我切换到本地模式继续..."
        \item "这个小问题正好展示了我们的错误处理机制..."
    \end{itemize}

    \vspace{0.3cm}

    \begin{alertblock}{切记}
    不要让技术问题影响整个展示，快速切换到备用方案
    \end{alertblock}
\end{frame}

\begin{frame}{Demo执行：讲解与同步}
    \textbf{边演示边讲解，让观众跟上思路}

    \vspace{0.3cm}

    \textbf{讲解同步技巧：}
    \begin{itemize}
        \item \textbf{预告下一步}：
            \begin{itemize}
                \item "接下来系统将进行..."
                \item "大家注意看这里的变化..."
            \end{itemize}

        \item \textbf{实时解释}：
            \begin{itemize}
                \item "现在系统正在检测边缘..."
                \item "这里可以看到定位结果..."
            \end{itemize}

        \item \textbf{突出重点}：
            \begin{itemize}
                \item "这个效果就是我们的创新点..."
                \item "注意这里处理速度的提升..."
            \end{itemize}

        \item \textbf{控制节奏}：
            \begin{itemize}
                \item 关键步骤放慢
                \item 简单操作快速带过
                \item 等待观众理解
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{Demo执行：总结收尾}
    \textbf{有力的结尾让Demo更完整}

    \vspace{0.3cm}

    \textbf{Demo总结要素：}
    \begin{enumerate}
        \item \textbf{回顾演示内容}：
            \begin{itemize}
                \item "刚才我们演示了..."
                \item "展示了X、Y、Z三个功能..."
            \end{itemize}

        \item \textbf{强调价值}：
            \begin{itemize}
                \item "整个过程只需3秒"
                \item "准确率达到95\%以上"
                \item "相比人工节省80\%时间"
            \end{itemize}

        \item \textbf{开放讨论}：
            \begin{itemize}
                \item "大家对哪个功能最感兴趣？"
                \item "有什么问题可以讨论"
            \end{itemize}
    \end{enumerate}

    \vspace{0.3cm}

    \begin{block}{完美Demo的标志}
    观众理解了功能价值，记住了核心亮点，提出了有意义的问题
    \end{block}
\end{frame}
