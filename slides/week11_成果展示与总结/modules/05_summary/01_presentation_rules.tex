\section{成果展示}

\begin{frame}{展示安排}
    \textbf{展示规则：}
    \begin{itemize}
        \item 每组展示：5分钟
        \item 答辩时间：2分钟
        \item 总时间：7分钟/组
    \end{itemize}

    \vspace{0.5cm}

    \textbf{展示顺序：} 抽签决定

    \vspace{0.5cm}

    \begin{alertblock}{严格控制时间！}
    时间到立即停止，确保所有小组都有展示机会
    \end{alertblock}

    \vspace{0.3cm}

    \textbf{展示内容要求：}
    \begin{itemize}
        \item 系统完整演示
        \item 技术实现说明
        \item 创新点展示
        \item 不足与改进方向
    \end{itemize}
\end{frame}
