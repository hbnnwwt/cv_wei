\begin{frame}{差异化展示策略}
    \textbf{根据项目复杂度和学生水平选择展示策略}

    \vspace{0.3cm}

    \textbf{基础展示策略（功能完善型）：}
    \begin{itemize}
        \item \textbf{重点}：展示完整功能流程
        \item \textbf{时间分配}：演示4分钟 + 讲解1分钟
        \item \textbf{成功标准}：功能完整运行、准确率达标、流程清晰
        \item \textbf{适合}：功能完善的系统、基础较弱的学生
    \end{itemize}

    \vspace{0.3cm}

    \textbf{进阶展示策略（技术创新型）：}
    \begin{itemize}
        \item \textbf{重点}：突出技术创新点
        \item \textbf{时间分配}：演示2.5分钟 + 讲解2.5分钟
        \item \textbf{成功标准}：展示1-2个技术创新、有实验数据支撑
        \item \textbf{适合}：有创新点的系统、技术能力强的学生
    \end{itemize}

    \vspace{0.3cm}

    \textbf{挑战展示策略（问题解决型）：}
    \begin{itemize}
        \item \textbf{重点}：讲述解决困难问题的过程
        \item \textbf{时间分配}：演示3分钟 + 故事2分钟
        \item \textbf{成功标准}：展示困难问题、解决过程清晰、有成长收获
        \item \textbf{适合}：遇到重大挑战的团队、善于讲故事的学生
    \end{itemize}
\end{frame}

\begin{frame}{面向不同听众的讲解策略}
    \textbf{如何向不同背景的人解释你的系统}

    \vspace{0.3cm}

    \textbf{向技术专家讲解：}
    \begin{itemize}
        \item 聚焦技术细节和算法选择
        \item 展示实验数据和性能对比
        \item 讨论技术挑战和解决方案
        \item 可以使用专业术语和公式
    \end{itemize}

    \vspace{0.3cm}

    \textbf{向非技术听众讲解：}
    \begin{itemize}
        \item 聚焦解决了什么问题、带来了什么价值
        \item 使用类比和故事解释技术概念
        \item 展示直观的Demo效果
        \item 避免术语，用生活化的语言
    \end{itemize}

    \vspace{0.3cm}

    \begin{block}{本周展示听众}
    \begin{itemize}
        \item \textbf{老师}：关注技术实现和学习收获
        \item \textbf{同学}：关注项目效果和经验分享
        \item 建议：平衡技术细节和价值展示
    \end{itemize}
    \end{block}
\end{frame}
