\begin{frame}{展示评分表}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{演示效果（30分）}
        \begin{itemize}
            \item 系统演示：15分
            \item 界面展示：10分
            \item 时间控制：5分
        \end{itemize}

        \vspace{0.3cm}

        \textbf{技术实现（40分）}
        \begin{itemize}
            \item 功能完整：15分
            \item 识别准确率：15分
            \item 代码质量：10分
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{答辩表现（20分）}
        \begin{itemize}
            \item 问题理解：10分
            \item 回答质量：10分
        \end{itemize}

        \vspace{0.3cm}

        \textbf{文档报告（10分）}
        \begin{itemize}
            \item 内容完整：10分
        \end{itemize}
    \end{columns}

    \vspace{0.5cm}

    \begin{block}{评分重点}
    \begin{itemize}
        \item 系统是否能正确识别答题卡
        \item 代码结构是否清晰
        \item 团队协作是否顺畅
        \item 创新点是否有价值
    \end{itemize}
    \end{block}
\end{frame}
