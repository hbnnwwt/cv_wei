\begin{frame}{为什么项目展示很重要？}
    \textbf{展示能力是工程师的核心软技能}

    \vspace{0.3cm}

    \begin{columns}
        \column{0.5\textwidth}
        \textbf{对个人发展：}
        \begin{itemize}
            \item \textbf{求职}：面试中展示项目经验
            \item \textbf{晋升}：向领导展示工作成果
            \item \textbf{影响力}：在团队中建立声誉
            \item \textbf{学习}：通过准备深化理解
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{对项目价值：}
        \begin{itemize}
            \item \textbf{获得反馈}：发现不足和改进点
            \item \textbf{推广价值}：让他人了解项目
            \item \textbf{积累声誉}：建立个人品牌
            \item \textbf{总结经验}：系统化梳理收获
        \end{itemize}
    \end{columns}

    \vspace{0.5cm}

    \begin{block}{展示能力的长期价值}
    \begin{itemize}
        \item 技术能力决定下限，展示能力决定上限
        \item 好的技术需要好的展示才能发挥价值
        \item 展示过程本身就是学习和成长
    \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{如何在展示中脱颖而出}
    \textbf{从合格到优秀的展示技巧}

    \vspace{0.3cm}

    \textbf{合格展示 vs 优秀展示：}
    \begin{table}
        \centering
        \small
        \begin{tabular}{p{5cm}p{5cm}}
            \toprule
            \textbf{合格展示} & \textbf{优秀展示} \\
            \midrule
            罗列功能清单 & 讲述问题和解决方案的故事 \\
            平铺直叙地演示 & 设计有起伏的演示流程 \\
            技术细节堆砌 & 聚焦核心价值和创新 \\
            被动回答问题 & 主动引导思考 \\
            单纯完成展示 & 激发兴趣和讨论 \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3cm}

    \textbf{脱颖而出的关键：}
    \begin{enumerate}
        \item \textbf{故事化}：用故事串联技术内容
        \item \textbf{价值导向}：始终聚焦"解决了什么问题"
        \item \textbf{真诚}：诚实面对不足，展示真实过程
        \item \textbf{热情}：传递对技术的热爱
    \end{enumerate}
\end{frame}

\begin{frame}{常见展示问题分析}
    \textbf{避免这些常见错误}

    \vspace{0.3cm}

    \textbf{内容层面：}
    \begin{itemize}
        \item[$\times$] 内容过多，信息过载
        \item[$\times$] 缺乏主线，逻辑混乱
        \item[$\times$] 过度技术，听众跟不上
        \item[$\times$] 没有故事，像念说明书
    \end{itemize}

    \vspace{0.2cm}

    \textbf{表达层面：}
    \begin{itemize}
        \item[$\times$] 声音太小，后排听不清
        \item[$\times$] 语速过快，没有停顿
        \item[$\times$] 照屏念稿，没有交流
        \item[$\times$] 肢体僵硬，缺乏自然
    \end{itemize}

    \vspace{0.2cm}

    \textbf{Demo层面：}
    \begin{itemize}
        \item[$\times$] 准备不足，现场翻车
        \item[$\times$] 流程混乱，反复跳转
        \item[$\times$] 没有备用方案
        \item[$\times$] 时间失控，虎头蛇尾
    \end{itemize}

    \vspace{0.3cm}

    \begin{alertblock}{预防措施}
    充分准备 + 多次排练 + 预设问题 + 备用方案
    \end{alertblock}
\end{frame}
