\subsection{综合实战：完整项目展示}

\begin{frame}{综合实战：展示准备清单}
    \textbf{完整项目展示准备}

    \vspace{0.3cm}

    \begin{columns}
        \column{0.5\textwidth}
        \textbf{内容准备：}
        \begin{itemize}
            \item[$\square$] 演示幻灯片（10-15页）
            \item[$\square$] Demo脚本与测试数据
            \item[$\square$] 备用方案（视频/截图）
            \item[$\square$] 技术文档（README）
            \item[$\square$] 代码仓库整理
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{硬件准备：}
        \begin{itemize}
            \item[$\square$] 笔记本电源
            \item[$\square$] 投影测试
            \item[$\square$] 翻页器/鼠标
            \item[$\square$] 备用U盘
        \end{itemize}
    \end{columns}

    \vspace{0.3cm}

    \textbf{演练准备：}
    \begin{itemize}
        \item[$\square$] 完整演练至少3次
        \item[$\square$] 计时并调整节奏
        \item[$\square$] 准备Q&A答案
        \item[$\square$] 预判技术问题
    \end{itemize}
\end{frame}

\begin{frame}{综合实战：完整展示流程}
    \textbf{7分钟完整展示流程}

    \vspace{0.3cm}

    \begin{table}
        \centering
        \small
        \begin{tabular}{llp{5cm}}
            \toprule
            \textbf{时间} & \textbf{环节} & \textbf{内容} \\
            \midrule
            0:00-0:30 & 开场介绍 & 组员、项目、核心功能 \\
            0:30-3:30 & 系统演示 & 完整识别流程、关键功能 \\
            3:30-4:30 & 技术讲解 & 架构、亮点、创新点 \\
            4:30-5:00 & 总结致谢 & 不足、展望、感谢 \\
            \midrule
            \textbf{5:00-7:00} & \textbf{Q\&A环节} & \textbf{回答评委问题} \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3cm}

    \begin{alertblock}{时间控制要点}
    \begin{itemize}
        \item 严格遵守5分钟展示时间
        \item 2分钟答辩要简明扼要
        \item 时间不够时先讲核心功能
        \item 提前30秒结束为Q&A留出时间
    \end{itemize}
    \end{alertblock}
\end{frame}

\begin{frame}{综合实战：Q&A准备}
    \textbf{预设常见问题与答案}

    \vspace{0.3cm}

    \begin{block}{技术类问题}
    \begin{itemize}
        \item Q: 为什么选择OpenCV而不是深度学习？
        \item A: OpenCV轻量、快速、适合传统图像处理...

        \item Q: 如何处理光照不均的情况？
        \item A: 使用自适应阈值和图像增强...
    \end{itemize}
    \end{block}

    \begin{block}{项目类问题}
    \begin{itemize}
        \item Q: 最大的挑战是什么？如何解决？
        \item A: 答题卡定位最困难，我们设计了...

        \item Q: 如何进一步提升准确率？
        \item A: 可以增加训练数据、优化算法...
    \end{itemize}
    \end{block}

    \begin{block}{团队类问题}
    \begin{itemize}
        \item Q: 团队如何分工协作？
        \item A: 我们按模块分工，每周同步进度...
    \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{综合实战：展示后总结}
    \textbf{从每次展示中学习}

    \vspace{0.3cm}

    \textbf{展示后立即做：}
    \begin{enumerate}
        \item \textbf{记录反馈}：
            \begin{itemize}
                \item 评委的提问
                \item 观众的反应
                \item 自己的感受
            \end{itemize}

        \item \textbf{总结亮点}：
            \begin{itemize}
                \item 哪些部分获得好评？
                \item 哪些展示方式有效？
                \item 哪些互动引起共鸣？
            \end{itemize}

        \item \textbf{找出不足}：
            \begin{itemize}
                \item 时间控制如何？
                \item 表达是否清晰？
                \item 技术问题如何应对？
            \end{itemize}

        \item \textbf{改进计划}：
            \begin{itemize}
                \item 下次如何做得更好？
                \item 需要加强哪方面准备？
                \item 有哪些可以复用？
            \end{itemize}
    \end{enumerate}
\end{frame}
