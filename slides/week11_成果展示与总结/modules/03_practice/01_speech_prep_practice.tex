\subsection{技术演讲准备实战}

\begin{frame}{演讲准备实战：确定主题与目标}
    \textbf{第一步：明确演讲目标}

    \vspace{0.3cm}

    \begin{block}{目标分析框架}
    \begin{itemize}
        \item \textbf{听众是谁？} 教师、同学、评委？
        \item \textbf{他们的背景？} 技术水平、专业知识
        \item \textbf{他们的期待？} 想听什么、学到什么
        \item \textbf{你的目标？} 展示成果、分享经验、获得反馈
    \end{itemize}
    \end{block}

    \vspace{0.3cm}

    \textbf{本周演讲目标：}
    \begin{itemize}
        \item 展示AI阅卷助手的完整功能
        \item 说明技术实现思路
        \item 分享项目开发经验
        \item 获得有价值的反馈
    \end{itemize}
\end{frame}

\begin{frame}{演讲准备实战：内容大纲设计}
    \textbf{第二步：构建内容大纲}

    \vspace{0.3cm}

    \begin{columns}
        \column{0.5\textwidth}
        \textbf{标准技术演讲结构：}
        \begin{enumerate}
            \item \textbf{Hook}（30秒）
                \begin{itemize}
                    \item 引人入胜的开场
                    \item 问题陈述
                \end{itemize}
            \item \textbf{Context}（1分钟）
                \begin{itemize}
                    \item 项目背景
                    \item 技术栈
                \end{itemize}
            \item \textbf{Demo}（3分钟）
                \begin{itemize}
                    \item 实际演示
                    \item 功能展示
                \end{itemize}
        \end{enumerate}

        \column{0.5\textwidth}
        \textbf{本周项目展示大纲：}
        \begin{enumerate}
            \item 开场介绍
                \begin{itemize}
                    \item 组员介绍
                    \item 项目名称
                    \item 核心功能一句话
                \end{itemize}
            \item 系统演示
                \begin{itemize}
                    \item 完整流程展示
                    \item 关键功能突出
                \end{itemize}
            \item 技术讲解
                \begin{itemize}
                    \item 架构设计
                    \item 创新亮点
                \end{itemize}
            \item 总结致谢
        \end{enumerate}
    \end{columns}
\end{frame}

\begin{frame}{演讲准备实战：故事线构建}
    \textbf{第三步：设计故事线}

    \vspace{0.3cm}

    \begin{block}{优秀技术演讲的故事线}
    \textbf{挑战} $\to$ \textbf{探索} $\to$ \textbf{突破} $\to$ \textbf{成果}

    \vspace{0.2cm}

    \begin{itemize}
        \item \textbf{挑战}：阅卷工作量大、易出错
        \item \textbf{探索}：调研CV技术、设计方案
        \item \textbf{突破}：解决定位、识别、集成难题
        \item \textbf{成果}：完整的AI阅卷助手
    \end{itemize}
    \end{block}

    \vspace{0.3cm}

    \textbf{避免的常见错误：}
    \begin{itemize}
        \item[$\times$] 简单罗列功能清单
        \item[$\times$] 过度技术细节堆砌
        \item[$\times$] 缺乏问题-解决方案的叙事
        \item[$\checkmark$] 围绕问题和解决方案讲故事
    \end{itemize}
\end{frame}

\begin{frame}{演讲准备实战：幻灯片制作}
    \textbf{第四步：制作幻灯片}

    \vspace{0.3cm}

    \textbf{设计原则：}
    \begin{itemize}
        \item \textbf{一页一观点}：每张幻灯片只传达一个核心信息
        \item \textbf{视觉化}：用图表代替文字，用代码截图代替描述
        \item \textbf{简洁}：去除不必要的装饰，聚焦内容
        \item \textbf{一致性}：统一风格、字体、配色
    \end{itemize}

    \vspace{0.3cm}

    \textbf{本周项目展示幻灯片建议：}
    \begin{table}
        \centering
        \small
        \begin{tabular}{lp{8cm}l}
            \toprule
            \textbf{页码} & \textbf{内容} & \textbf{时间} \\
            \midrule
            1 & 封面：项目名称、组员 & 10秒 \\
            2 & 项目概述：一句话说明功能 & 20秒 \\
            3-4 & 系统架构图、技术栈 & 30秒 \\
            5-8 & 核心功能演示截图/GIF & 2分钟 \\
            9 & 技术亮点 & 30秒 \\
            10 & 不足与展望 & 20秒 \\
            \bottomrule
        \end{tabular}
    \end{table}
\end{frame}

\begin{frame}{演讲准备实战：时间控制与排练}
    \textbf{第五步：时间控制与排练}

    \vspace{0.3cm}

    \textbf{5分钟展示的时间分配：}
    \begin{table}
        \centering
        \begin{tabular}{lcl}
            \toprule
            \textbf{环节} & \textbf{时间} & \textbf{提示} \\
            \midrule
            开场介绍 & 30秒 & 快速进入主题 \\
            系统演示 & 3分钟 & 核心环节，留足时间 \\
            技术讲解 & 1分钟 & 聚焦亮点，不求全面 \\
            总结致谢 & 30秒 & 简洁有力 \\
            \midrule
            \textbf{总计} & \textbf{5分钟} & 严格控制！ \\
            \bottomrule
        \end{tabular}
    \end{table}

    \vspace{0.3cm}

    \textbf{排练要点：}
    \begin{itemize}
        \item 至少完整排练3次
        \item 计时并记录每个环节用时
        \item 准备1分钟精简版（以防时间紧张）
        \item 邀请同学作为试听观众
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{演讲准备实战：开场白设计}
    \textbf{设计一个吸引人的开场}

    \vspace{0.3cm}

    \begin{block}{开场白模板}
    \begin{itemize}
        \item \textbf{问候}：大家好/各位老师、同学好
        \item \textbf{团队}：我们是XX组，成员有A、B、C、D
        \item \textbf{项目}：今天展示的项目是"AI阅卷助手"
        \item \textbf{Hook}：你是否曾为期末阅卷的繁琐而烦恼？
        \item \textbf{价值}：我们的系统能自动识别答题卡，准确率达XX\%
        \item \textbf{预告}：接下来让我演示完整功能
    \end{itemize}
    \end{block}

    \vspace{0.3cm}

    \begin{exampleblock}{示例开场}
    \small
    "大家好！我们是'阅卷先锋'组。面对成堆的答题卡，手动阅卷既费时又易出错。我们开发的AI阅卷助手，能自动识别选择题和判断题，准确率达到95\%以上。接下来让我演示它是如何工作的..."
    \end{exampleblock}
\end{frame}
