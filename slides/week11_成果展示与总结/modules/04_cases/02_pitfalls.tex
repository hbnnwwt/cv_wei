\begin{frame}{常见展示问题分析}
    \textbf{识别并避免这些常见问题}

    \vspace{0.3cm}

    \begin{columns}
        \column{0.5\textwidth}
        \textbf{问题1：内容组织混乱}
        \begin{itemize}
            \item \textit{表现}：
                \begin{itemize}
                    \item 跳跃式讲述
                    \item 缺乏主线
                    \item 信息过载
                \end{itemize}
            \item \textit{原因}：
                \begin{itemize}
                    \item 没有提前设计结构
                    \item 想讲的太多
                \end{itemize}
            \item \textit{对策}：
                \begin{itemize}
                    \item 设计清晰大纲
                    \item 聚焦核心信息
                    \item 一个观点一张幻灯片
                \end{itemize}
        \end{itemize}

        \column{0.5\textwidth}
        \textbf{问题2：时间控制不当}
        \begin{itemize}
            \item \textit{表现}：
                \begin{itemize}
                    \item 前松后紧
                    \item 超时或过早结束
                    \item 关键内容没时间讲
                \end{itemize}
            \item \textit{原因}：
                \begin{itemize}
                    \item 没有排练
                    \item 临时发挥
                \end{itemize}
            \item \textit{对策}：
                \begin{itemize}
                    \item 多次排练计时
                    \item 准备精简版
                    \item 设计可跳过内容
                \end{itemize}
        \end{itemize}
    \end{columns}
\end{frame}

\begin{frame}{常见展示问题分析（续）}
    \vspace{0.3cm}

    \textbf{问题3：技术故障处理不当}
    \begin{itemize}
        \item \textit{表现}：
            \begin{itemize}
                \item 现场反复尝试修复
                \item 慌张失措
                \item 浪费大量时间
            \end{itemize}
        \item \textit{原因}：
            \begin{itemize}
                \item 没有备用方案
                \item 环境没有测试
            \end{itemize}
        \item \textit{对策}：
            \begin{itemize}
                \item 准备演示视频
                \item 提前测试环境
                \item 快速切换备用方案
            \end{itemize}
    \end{itemize}

    \vspace{0.3cm}

    \textbf{问题4：Q&A应对失当}
    \begin{itemize}
        \item \textit{表现}：
            \begin{itemize}
                \item 防御性回答
                \item 胡乱回答
                \item 陷入争论
            \end{itemize}
        \item \textit{对策}：
            \begin{itemize}
                \item 诚实面对不知道的问题
                \item 保持开放态度
                \item 引导到核心价值
            \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{预防策略总结}
    \textbf{如何避免这些常见问题？}

    \vspace{0.3cm}

    \begin{block}{展示前}
    \begin{itemize}
        \item 设计清晰的结构大纲
        \item 多次排练并计时
        \item 准备备用方案
        \item 预设问题答案
    \end{itemize}
    \end{block}

    \begin{block}{展示中}
    \begin{itemize}
        \item 严格控制时间节奏
        \item 保持冷静和微笑
        \item 出问题时快速切换
        \item 诚实面对不知道的问题
    \end{itemize}
    \end{block}

    \begin{block}{展示后}
    \begin{itemize}
        \item 记录观众反馈
        \item 总结亮点和不足
        \item 为下次展示改进
    \end{itemize}
    \end{block}
\end{frame}
