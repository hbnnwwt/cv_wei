\begin{frame}{课堂互动：场景模拟}

    \vspace{0.3cm}

    \textbf{请选择最佳应对方式：}
    \begin{itemize}
        \item[$\square$] A. 现场调试，直到修复为止
        \item[$\square$] B. 假装没看到，继续讲解
        \item[$\square$] C. 立即切换到备用视频
        \item[$\square$] D. 取消Demo，只讲PPT
    \end{itemize}

    \pause

    \begin{exampleblock}{最佳答案：C}
    现场调试浪费时间和展示机会，应该立即切换到备用方案，保持演示流畅
    \end{exampleblock}

    \vspace{0.3cm}

    \textbf{可选话术：}
    \begin{itemize}
        \item "看来我们的AI今天有点害羞，让我用录制好的视频来展示..."
        \item "为了节省时间，我切换到预先录制的演示..."
    \end{itemize}
\end{frame}
