\begin{frame}{阅卷系统展示常见问题}
    \textbf{避免这些错误}

    \vspace{0.3cm}

    \textbf{内容问题：}
    \begin{itemize}
        \item[$\times$] 演示时间过长，留给讲解时间不足
        \item[$\times$] 功能堆砌，没有重点和亮点
        \item[$\times$] 过度强调技术细节，忽略价值展示
        \item[$\times$] Demo准备不足，现场出现bug
    \end{itemize}

    \vspace{0.2cm}

    \textbf{表达问题：}
    \begin{itemize}
        \item[$\times$] 只是念PPT，没有互动
        \item[$\times$] 技术术语过多，听众跟不上
        \item[$\times$] 时间失控，被老师叫停
        \item[$\times$] 答辩准备不足，答不上问题
    \end{itemize}

    \vspace{0.3cm}

    \textbf{最佳实践：}
    \begin{itemize}
        \item[$\checkmark$] 提前录制演示视频，避免现场翻车
        \item[$\checkmark$] 聚焦1-2个核心亮点，不求面面俱到
        \item[$\checkmark$] 用数据和结果证明效果
        \item[$\checkmark$] 准备备用演示方案
    \end{itemize}
\end{frame}

\begin{frame}{阅卷系统展示技巧}
    \textbf{让阅卷系统展示更出彩}

    \vspace{0.3cm}

    \textbf{突出核心价值：}
    \begin{itemize}
        \item "传统阅卷需要30分钟，我们只需30秒"
        \item "准确率从人工的90\%提升到95\%"
        \item "支持5种不同格式的答题卡"
    \end{itemize}

    \vspace{0.3cm}

    \textbf{展示技术亮点：}
    \begin{itemize}
        \item \textbf{定位算法}：自适应锚点定位，成功率98\%
        \item \textbf{识别算法}：多特征融合填涂检测
        \item \textbf{容错机制}：异常情况自动处理
    \end{itemize}

    \vspace{0.3cm}

    \textbf{讲述开发故事：}
    \begin{itemize}
        \item 遇到的问题：光照不均、填涂浅、答题卡偏移
        \item 解决的过程：尝试多种方案、实验对比
        \item 团队的成长：从不会到会的转变
    \end{itemize}
\end{frame}
