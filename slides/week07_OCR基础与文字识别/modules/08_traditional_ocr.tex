\section{传统OCR识别实战}

\begin{frame}{传统OCR识别流程概述}
    \begin{columns}
        \column{0.5\textwidth}
        \textbf{传统OCR四步走：}
        \begin{enumerate}
            \item \textbf{预处理}：图像增强、二值化、去噪
            \item \textbf{版面分析}：文字区域定位、分栏
            \item \textbf{字符分割}：行分割、字分割
            \item \textbf{字符识别}：特征提取、分类器
        \end{enumerate}

        \column{0.5\textwidth}
        \textbf{各阶段关键技术：}
        \begin{itemize}
            \item 预处理：灰度化、二值化、倾斜校正
            \item 版面分析：投影分析、连通域分析
            \item 分割：投影分割、连通域分割
            \item 识别：模板匹配、特征分类
        \end{itemize}
    \end{columns}
\end{frame}
