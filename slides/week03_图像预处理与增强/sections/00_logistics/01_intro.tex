\section{为什么需要预处理？}

\begin{frame}{预处理是计算机视觉的"门卫"}
    \begin{columns}
        \column{0.5\textwidth}
        \begin{alertblock}{现实问题}
            \begin{itemize}
                \item 拍摄角度不正
                \item 光照不均匀
                \item 纸张有折痕
                \item 背景有杂物
                \item \textbf{手机拍照 vs 扫描仪}
            \end{itemize}
        \end{alertblock}

        \column{0.5\textwidth}
        \begin{exampleblock}{预处理目标}
            \begin{enumerate}
                \item 去除噪声干扰
                \item 增强目标特征
                \item 规范图像格式
                \item \textbf{提升识别准确率}
            \end{enumerate}
        \end{exampleblock}
    \end{columns}

    \vspace{0.5cm}
    \centering
    \textbf{\textcolor{red}{预处理质量直接决定后续识别效果！}}
\end{frame}
