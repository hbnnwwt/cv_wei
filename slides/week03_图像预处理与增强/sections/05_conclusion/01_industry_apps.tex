\begin{frame}{扫描全能王的预处理技术}
    \textbf{核心技术：}
    \begin{itemize}
        \item \textbf{自动边缘检测}：智能识别文档边界
        \item \textbf{透视矫正}：自动展平拍摄变形
        \item \textbf{多帧融合}：多张拍摄降噪
        \item \textbf{智能增强}：针对不同场景调优
    \end{itemize}

    \vspace{0.3cm}

    \textbf{处理流程：}
    \begin{center}
        边缘检测 $\rightarrow$ 透视矫正 $\rightarrow$ 去噪 $\rightarrow$ 增强 $\rightarrow$ 压缩
    \end{center}
\end{frame}

\begin{frame}{银行票据自动处理}
    \textbf{应用场景：}
    \begin{itemize}
        \item 支票识别
        \item 发票处理
        \item 表单录入
    \end{itemize}

    \vspace{0.3cm}

    \textbf{预处理要求：}
    \begin{itemize}
        \item \textbf{高精度}：错误容忍度极低
        \item \textbf{标准化}：统一输入格式
        \item \textbf{鲁棒性}：应对各种拍摄条件
        \item \textbf{可追溯}：保留处理日志
    \end{itemize}

    \vspace{0.3cm}

    \textbf{特殊技术：}
    \begin{itemize}
        \item 红外去伪（防伪检测）
        \item 水印去除
        \item 印章分离
    \end{itemize}
\end{frame}

\begin{frame}{档案数字化预处理}
    \textbf{挑战：}
    \begin{itemize}
        \item 纸张老化、发黄
        \item 字迹褪色
        \item 装订孔遮挡
        \item 背面透字
    \end{itemize}

    \vspace{0.3cm}

    \textbf{解决方案：}
    \begin{itemize}
        \item \textbf{背景白化}：去除纸张底色
        \item \textbf{字迹增强}：对比度拉伸
        \item \textbf{去网纹}：去除印刷网纹
        \item \textbf{装订修复}：Inpainting 填补
    \end{itemize}
\end{frame}
